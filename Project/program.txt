func largest(x,y,z)
{
    ## Braces are required in all functions, loops, and conditionals.
    if(x > y)
    {
        if(x > z)
        {
            return x;
        }
        return z;
    }
    else if (y > x)
    {
        if(y > z)
        {
            return y;
        }
        return z;
    }
    else
    {
        if(y > z)
        {
            return y;
        }
        return z;
    }
}

func
apply(f, L, size)
{
    var x = L[0];
    while (x != L[size])
    {
        f(L[x]);
        var x = L[x + 1];
    }
}
